#include <iostream>
#include <vector>
#include <numeric>
using namespace std;

int main() {
    vector<int> numbers;
    int n;
    cin >> n;

    for(int i = 0; i < n; ++i) {
        int item;
        cin >> item;
        numbers.push_back(item);
    }

    int sum = accumulate(numbers.begin(), numbers.end(), 0);
    cout << sum; 

    return 0;
}
